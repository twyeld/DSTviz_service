[old] docker build -t nvidia-base-game-app-xorg-gl .
[older] docker build -t nvidia-base-game-app-xorg .

to run:
x11docker --hostdisplay --gpu nvidia-base-game-app-xorg-gl-unity xterm
[old] x11docker --hostdisplay --gpu nvidia-base-game-app-xorg-gl xterm
[older] x11docker --hostdisplay --gpu nvidia-base-game-app-xorg xterm

cd home/app

./<my_game.x86_64>
